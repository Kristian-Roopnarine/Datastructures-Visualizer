class LinkedList {
    constructor(r=null){
        this.root = r
        this.size =0
    }
}

    constructor(value){
        this.value=value
        this.next=null
    }
}

// Linked List class method to add Node instances
add(node){
    if (this.root === null){
        this.root = node
        return
    } else {
        node.next = this.root
        this.root = node
    }
    this.size++
}
                        
                    
                        
// Linked List class method

find(value){
    let thisNode = this.root
    while (thisNode !== null){
        if (thisNode.value === value){
            return true
        } else {
            thisNode = thisNode.next
        }
    }
    return false
}

                        
                    
                        
// Linked List class method

delete(value){
    let thisNode = this.root
    let prevNode = null

    while (thisNode !== null){
        if (thisNode.value === value){
            if (prevNode === null){
                this.root = thisNode.next
                thisNode.next = null
                return
            }
            prevNode.next = thisNode.next
            thisNode.next = null
            return

        } else {
            prevNode = thisNode
            thisNode = thisNode.next
        }
    }

    return false
}

                        
                    
                        
// Linked List class method

insertBefore(valueToFind,nodeToInsert){
    let thisNode = this.root
    let prevNode = null

    while (thisNode !== null) {
        if (valueToFind === thisNode.value){
            if (prevNode === null){
                this.add(nodeToInsert)
            } else {
                nodeToInsert.next = prevNode.next
                prevNode.next = nodeToInsert
                return
            }
        } else {
            prevNode = thisNode
            thisNode = thisNode.next
        }
    }
    return false
}
